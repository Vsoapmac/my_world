自带数据结构头文件/库/包大总结

无论是c++、java、还是python，都会自带已经实现的数据结构，下面是总结所有的数据结构的头文件/库/包的用法

C++

无论引用任何的数据结构头文件，都需要加上以下命名空间

using namespace std;

双向链表

引用

#include <list>

初始化

list<int> _list;// 一个空的节点为int类型的链表
list<int> _list(10);// 一个长度为10的节点为int类型的链表

插入数据

从前面插入一个数据

_list.push_front(1);

从后面插入一个数据

_list.push_back(2)

移除数据

移除与值相同的节点

_list.remove(1);

移除最前面的数据

_list.pop_front();

移除最后面的数据

_list.pop_back();

清空链表的内容

_list.clear();

排序

_list.sort();

查找

查看链表最前面的元素

_list.front();

遍历所有的list元素

list<int>::iterator iter;
for(iter = _list.begin();iter!= _list.end();iter++){
    cout << *iter << endl;
}

查看链表的长度

_list.size();

查看链表是否为空

_list.empty();

队列

引用

#include <queue>

初始化

queue<int> _queue;// int类型的队列

入队

_queue.push(1);

访问队头元素

_queue.front();

队头出队

_queue.pop();

检查队伍是否为空

_queue.empty();

查看队列长度

_queue.size();

栈

引用

#include <stack>

初始化

stack<int> _stack;// int类型的栈

入栈

_stack.push(1);

查看栈顶元素

int top = _stack.top();

将栈顶元素移出栈

_stack.pop();

查看栈大小

_stack.size();

JAVA

链表

插入效率高

引用

import java.util.LinkedList;

初始化

LinkedList<Integer> list = new LinkedList<>();

添加元素

添加

list.add(1)

从头部添加

list.addFirst(2)

从尾部添加

list.addLast(3)

删除元素

删除头部元素

list.removeFirst();

删除尾部元素

list.removeLast();

查看链表长度

list.size();

查找元素

获取头部元素

list.getFirst();

获取尾部元素

list.getLast();

获取指定位置的元素

list.get(0); // index为0的元素

迭代查找（遍历）

for (int i : list) {
    System.out.println(i);
}

可动态修改数组（非数据结构）

查找效率高

引用

import java.util.ArrayList;

初始化

ArrayList<int> list =new ArrayList<>();

添加元素

list.add(1)

查找元素

获取特定位置的元素

list.get(0); // index为0的元素

迭代

for (int i : list) {
    System.out.println(i);
}

修改元素

list.set(0, 1); // index为0的元素修改为1

删除元素

删除特定位置的元素

list.remove(1); // 删除index为1的元素

删除所有

list.clear();

查看长度

list.size();

排序

import java.util.Collections;  // 引入 Collections 类
Collections.sort(list);  // 排序

队列

LinkedList实现了queue接口

引用

import java.util.LinkedList;
import java.util.Queue;

初始化

LinkedList<Integer> queue = new LinkedList<>();
Queue<Node> queue = new LinkedList<>();

添加元素

queue.offer(1);

取队头元素

int peek=queue.peek();

队头出队

int top = queue.pop();
int top = queue.poll();// 用Queue的时候用这个方法

判断是否为空

queue.isEmpty(); // 为空返回T,否则为F

栈

引用

import java.util.Stack;

初始化

Stack<Integer> stack=new Stack<Integer>();

取栈顶元素

int top = stack.peek();

栈顶元素出站并获取

int top=stack.pop();

入栈

stack.push(1);

判断栈是否为空

stack.empty();
